#!/bin/bash
PREFIX=/home/ccscanf/src/
for dir in $(ls -l | grep "^d"|awk ' { print $9 } ')
do
  cd $PREFIX$dir;
  read -p "Would you like to delete [$dir]?(n/y)" -n 1 -r
  git status;
  if [[ $REPLY =~ ^[Yy]$ ]]
  then
    rm -rvf "$PREFIX$dir"
  fi
  echo
  echo
  clear
done;
